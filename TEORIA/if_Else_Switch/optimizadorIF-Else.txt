static void Main(string[] args)
        {
            //Declaración de variables
            double a, b, resultado = 0.0;
            int opcion = 0;

            //Mostrar el menú
            Console.WriteLine(" 1. Suma");
            Console.WriteLine(" 2. Resta");
            Console.WriteLine(" 3. Divsión");
            Console.WriteLine(" 4. Multiplicación");

            Console.Write("Elige una opción: ");
            opcion = Convert.ToInt32(Console.ReadLine());

            //Pedimos el primer número
            Console.Write("Dame el primer número: ");
            a = Convert.ToDouble(Console.ReadLine());

            //Pedimos el segundo número
            Console.Write("Dame el segundo número: ");
            b = Convert.ToDouble(Console.ReadLine());

            if (opcion == 1)//SUMA
            {
                resultado = a + b;
            }
            else
            {
                if (opcion == 2)//RESTA
                {
                    resultado = a - b;
                }
                else
                {
                    if (opcion == 3)//DIVISIÓN
                    {
                        if (b != 0)//Este if esta anidado
                            resultado = a / b;
                        else //Este else pertenece al if anidado
                        {
                            Console.WriteLine("El divisior no es valido");
                        }
                    }
                    else
                    {
                        if (opcion == 4)//MULTIPLICACIÓN
                        {
                            resultado = a * b;
                        }
                    }
                }
            }

            //Mostramos el resultado
            Console.WriteLine("El resultado es: {0}", resultado);

        }
